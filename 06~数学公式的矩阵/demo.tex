    % 矩阵环境，用&分割列，用\\分割行
    \[
    \begin{matrix}
        0 & 1 \\
        1 & 0
    \end{matrix} 
    % pmatrix环境 小括号
    \begin{pmatrix}
        0 & 1 \\
        1 & 0   
    \end{pmatrix}
    % Vmatrix环境 ||
    \begin{Vmatrix}
        0 & 1 \\
        1 & 0
    \end{Vmatrix}
    \]

    \[
      A = \begin{pmatrix}
          a_{11}^2 & a_{12}^2 & a_{13}^2 \\
          0 & a_{22} & a_{23} \\
          0 & 0 & a_{33}
      \end{pmatrix}  
    \]
      % \times 是排版乘号 bmatrix环境 是 中括号
    \[
      A = \begin{bmatrix}
          a_{11} & \dots & a_{1n} \\
          & \ddots & \vdots \\
          0 & & a_{nn}
      \end{bmatrix}_{n \times n}  
    \]

    
    \[
      \begin{pmatrix}
          \begin{matrix}
              1 & 0 \\ 0 & 1
          \end{matrix}
          & \text{\Large 0 } & \begin{matrix}
              1& 0 \\ 0 & -1 
          \end{matrix}
      \end{pmatrix} 
    \]
      % 跨列的省略号 ： \hdotsfor{列数}
    \[
      \begin{pmatrix}
          1& \frac 12 & \dots & \frac 1n \\
          \hdotsfor{4} \\
          m & \frac m2 & \dots & \frac mn 
      \end{pmatrix}
    \]
      % 行内小矩阵 （smallmatrix） 环境
    负数 $z= (x,y) $ 也可以用矩阵
    \begin{math}
        \left( % 需要手动加上左括号
            \begin{smallmatrix}
                x & -y \\ y & x 
            \end{smallmatrix}
        \right) % 需要手动加上右括号
    \end{math}来表示


    \[
      \begin{array}{r|r}
          \frac12 & 0 \\
          \hline
          0 & -\frac abc
      \end{array}  
    \]
